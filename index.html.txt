<script>

let ocorrencias = JSON.parse(localStorage.getItem("bopm")) || []

let editandoID = null

function show(t){

document.getElementById("registro").style.display="none"
document.getElementById("consulta").style.display="none"

document.getElementById(t).style.display="block"

}

function registrar(){

let ocorrencia = {

id: editandoID || Date.now(),

natureza:natureza.value,
unidade:unidade.value,
local:local.value,
data:data.value,
policiais:policiais.value,
suspeito:suspeito.value,
relato:relato.value,
artigos:artigos.value,
observacao:observacao.value

}

if(editandoID){

let index = ocorrencias.findIndex(o => o.id === editandoID)

ocorrencias[index] = ocorrencia

editandoID = null

botaoRegistro.innerText = "Registrar Ocorrência"

}else{

ocorrencias.push(ocorrencia)

}

localStorage.setItem("bopm",JSON.stringify(ocorrencias))

limpar()

render()

alert("Ocorrência salva")

}

function abrir(id){

let o = ocorrencias.find(o => o.id === id)

natureza.value = o.natureza
unidade.value = o.unidade
local.value = o.local
data.value = o.data
policiais.value = o.policiais
suspeito.value = o.suspeito
relato.value = o.relato
artigos.value = o.artigos
observacao.value = o.observacao

editandoID = id

botaoRegistro.innerText = "Salvar Alterações"

show("registro")

}

function copiarBoletim(){

let texto = `
=============================
BOLETIM DE OCORRÊNCIA PMESP
=============================

Natureza da Ocorrência:
${natureza.value}

Unidade:
${unidade.value}

Local:
${local.value}

Data:
${data.value}

Policiais Envolvidos:
${policiais.value}

Suspeito:
${suspeito.value}

Relato:
${relato.value}

Artigos Cabíveis:
${artigos.value}

Observação:
${observacao.value}

=============================
Sistema BOPM
Polícia Militar do Estado de São Paulo
=============================
`

navigator.clipboard.writeText(texto)

alert("Boletim copiado")

}

function limpar(){

natureza.value=""
unidade.value=""
local.value=""
data.value=""
policiais.value=""
suspeito.value=""
relato.value=""
artigos.value=""
observacao.value=""

}

function render(){

let tabela=document.getElementById("tabela")

tabela.innerHTML=""

ocorrencias.forEach(o=>{

tabela.innerHTML+=`

<tr>
<td>${o.id}</td>
<td>${o.natureza}</td>
<td>${o.local}</td>
<td>${o.suspeito}</td>
<td>
<span class="action" onclick="abrir(${o.id})">Abrir</span>
</td>
</tr>

`

})

}

render()

</script>
